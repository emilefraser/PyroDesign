015    package com.google.api.gwt.client.impl;
016    
017    import com.google.api.gwt.client.OAuth2Login;
018    import com.google.api.gwt.shared.AuthScope;
019    import com.google.gwt.core.client.Callback;
020    import com.google.web.bindery.requestfactory.shared.Receiver;
021    import com.google.web.bindery.requestfactory.shared.ServerFailure;
022    
023    import java.util.Arrays;
024    
025    /**
026     * Handles the OAuth2 login process for GWT client code.
027     *
028     * @deprecated Use {@link OAuth2Login}
029     */
030    @Deprecated
031    public class ClientOAuth2Login implements com.google.api.gwt.shared.OAuth2Login<ClientOAuth2Login> {
032      private final String clientId;
033      private Iterable<AuthScope> scopes;
034    
035      public ClientOAuth2Login(String clientId) {
036        this.clientId = clientId;
037      }
038    
039      @Override
040      public ClientOAuth2Login withScopes(AuthScope... scopes) {
041        this.scopes = Arrays.asList(scopes);
042        return this;
043      }
044    
045      @Override
046      public void login(final Receiver<String> receiver) {
047        OAuth2Login.get().authorize(clientId, scopes, new Callback<Void, Exception>() {
048          @Override
049          public void onSuccess(Void v) {
050            receiver.onSuccess("");
051          }
052    
053          @Override
054          public void onFailure(Exception e) {
055            receiver.onFailure(new ServerFailure(e.getMessage()));
056          }
057        });
058      }
059    }
